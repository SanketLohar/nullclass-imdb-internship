
const { getActorById } = require('./src/data/actors/actor.service');

async function testActor() {
    console.log("Testing actor fetch for ID: 3002581");
    try {
        const actor = await getActorById(3002581, { strict: false });
        console.log("Result:", actor ? "Found" : "Null");
        if (actor) {
            console.log("Name:", actor.name);
            console.log("Bio length:", actor.biography?.length);
            console.log("Filmography count:", actor.filmography?.length);
        }
    } catch (e) {
        console.error("Error:", e);
    }
}
